167 size_t Socket::read(unsigned char buf[], size_t length)
168  {
169  if(sockfd == invalid_socket)
170  throw std::runtime_error("Socket::read: Socket not connected");
171 
172  size_t got = 0;
173 
174  while(length)
175  {
176  ssize_t this_time = ::recv(sockfd, (char*)buf + got,
177  length, MSG_NOSIGNAL);
178 
179  if(this_time == 0)
180  break;
181 
182  if(this_time == -1)
183  {
184  if(socket_error_code == EINTR)
185  this_time = 0;
186  else
187  throw std::runtime_error("Socket::read: Socket read failed");
188  }
189 
190  got += this_time;
191  length -= this_time;
192  }
193  return got;
194  }
